type expr =
| EUnit
| EConst of const
| EBinop of binop * expr * expr
| EVar of name
| EIf of expr * expr * expr
| ELet of decl * expr
| EFun of pattern * expr
| EApp of expr * expr
| EMatch of expr * (pattern * expr) list
| ECallM of name * name
| EObj of obj
and objexpr =
| OMeth of pattern * expr
| OVal of pattern * expr
and obj = objexpr list
and pattern =
| PUnit
| PVar of name
| PConst of const
and decl = bool * pattern * expr
and declaration =
| DLet of decl
