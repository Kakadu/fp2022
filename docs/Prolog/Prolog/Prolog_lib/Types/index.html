type choicepoint =
| Choicepoint of {
goal : Ast.term;
substitution : unifier;
candidates : clause list;
}
| Conjuction of {
goal : Ast.term;
choicepoints : choicepoints;
substitution : unifier;
}
| Clause of {
goal : Ast.term;
substitution : unifier;
candidates : clause list;
head : Ast.term;
body : Ast.term;
}

Defines three possible types of a choicepoint. Type of a choicepoints depends on the predicate while executing which it was created.
